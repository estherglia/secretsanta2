    <script>

        // --- Game Setup ---

        const allParticipants = [

            "Ron Mantel", "Kendra Bernstein", "Swathi Krothapalli", "Sabrina Dobbs", "Omar Garcia", 

            "Sean Stallings", "Esther Ly", "Kate Elisha", "Ebony Dixon", "Jillian DiPaolo"

        ];


        // Utility: Fisher-Yates Shuffle

        function shuffle(array) {

            let currentIndex = array.length, randomIndex;

            while (currentIndex != 0) {

                randomIndex = Math.floor(Math.random() * currentIndex);

                currentIndex--;

                [array[currentIndex], array[randomIndex]] = [array[randomIndex], array[currentIndex]];

            }

            return array;

        }


        // --- Modal Logic ---

        const modal = document.getElementById("gift-modal");

        const modalImg = document.getElementById("modal-img");

        const modalGiftName = document.getElementById("modal-gift-name");

        const modalFrom = document.getElementById("modal-from");

        const modalLink = document.getElementById("modal-link").firstElementChild;

        const closeModal = document.getElementById("modal-close");


        closeModal.onclick = () => modal.style.display = "none";

        window.onclick = (event) => {

            if (event.target == modal) {

                modal.style.display = "none";

            }

        }


        // --- Main Game Logic ---

        window.addEventListener('load', function() {

            const gifts = JSON.parse(localStorage.getItem('secretSantaGifts')) || [];

            

            if (gifts.length < allParticipants.length) {

                document.getElementById('instructions').innerHTML = 

                    `<strong style="color: #f00;">WAIT! Not all gifts have been submitted.</strong><br>

                    Expected ${allParticipants.length} gifts, but only found ${gifts.length}.<br>

                    Please check your 'admin.html' page.`;

                return;

            }


            const shuffledGifts = shuffle(gifts);

            const grid = document.getElementById('gift-grid');


            shuffledGifts.forEach((gift, index) => {

                const box = document.createElement('div');

                box.classList.add('box');

                box.innerHTML = `<span>${index + 1}</span>`;

                box.dataset.giftIndex = index; // Store the index of the gift


                box.addEventListener('click', openBox);

                grid.appendChild(box);

            });


            function openBox(e) {

                const box = e.currentTarget;

                if (box.classList.contains('opened')) {

                    alert('This gift has already been opened!');

                    return;

                }

                

                // Mark as opened

                box.classList.add('opened');


                // Get gift data

                const giftIndex = box.dataset.giftIndex;

                const gift = shuffledGifts[giftIndex];


                // Populate and show modal

                modalImg.src = gift.img;

                modalGiftName.textContent = gift.giftName;

                modalFrom.textContent = `Brought by: ${gift.from}`;

                modalLink.href = gift.link;

                

                modal.style.display = "block";

            }

        });
