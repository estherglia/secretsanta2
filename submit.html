    <script>

        document.getElementById('giftForm').addEventListener('submit', function(e) {

            e.preventDefault(); // Stop the form from submitting normally


            const newGift = {

                from: document.getElementById('yourName').value,

                giftName: document.getElementById('giftName').value,

                link: document.getElementById('giftLink').value,

                img: document.getElementById('imageLink').value

            };


            // Get existing gifts from localStorage, or create a new array

            let gifts = JSON.parse(localStorage.getItem('secretSantaGifts')) || [];


            // Check if this person has already submitted

            const existingSubmission = gifts.find(gift => gift.from === newGift.from);

            if (existingSubmission) {

                if (!confirm('You have already submitted a gift. Do you want to overwrite it?')) {

                    return; // Stop if they click "Cancel"

                }

                // Remove old submission

                gifts = gifts.filter(gift => gift.from !== newGift.from);

            }

            

            // Add the new gift

            gifts.push(newGift);


            // Save the updated list back to localStorage

            localStorage.setItem('secretSantaGifts', JSON.stringify(gifts));


            // Show success message and hide form

            document.getElementById('giftForm').style.display = 'none';

            document.getElementById('success-message').style.display = 'block';

        });

    </script>
